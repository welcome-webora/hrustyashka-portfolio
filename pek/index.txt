<script>
  const DELIVERY_PRICE = 200;
  const cart = {};
  const cartList = document.getElementById('cart-list');
  const totalEl = document.getElementById('total');
  const showCartBtn = document.getElementById('show-cart');
  const cartModal = document.getElementById('cart-modal');
  const closeCart = document.getElementById('close-cart');
  const orderForm = document.getElementById('order-form');
  const orderHidden = document.getElementById('order-hidden');

  function updateCart() {
    cartList.innerHTML = '';
    let total = 0;

    Object.keys(cart).forEach(name => {
      const item = cart[name];
      total += item.price * item.count;

      const li = document.createElement('li');
      li.innerHTML = `
        ${item.name} - ${item.price * item.count} ₽
        <div class="cart-controls">
          <button onclick="changeCount('${name}', -1)">-</button>
          <span>${item.count}</span>
          <button onclick="changeCount('${name}', 1)">+</button>
        </div>`;
      cartList.appendChild(li);
    });

    totalEl.textContent = total + DELIVERY_PRICE;
  }

  function changeCount(name, delta) {
    if (!cart[name]) return;
    cart[name].count += delta;
    if (cart[name].count <= 0) delete cart[name];
    updateCart();
  }

  document.querySelectorAll('.product').forEach(productEl => {
    const name = productEl.dataset.name;
    const price = parseInt(productEl.dataset.price);
    const minusBtn = productEl.querySelector('.minus');
    const plusBtn = productEl.querySelector('.plus');
    const countEl = productEl.querySelector('.count');
    let count = 0;

    plusBtn.addEventListener('click', () => {
      count++;
      countEl.textContent = count;
      cart[name] = {name, price, count};
      updateCart();
    });

    minusBtn.addEventListener('click', () => {
      if (count > 0) {
        count--;
        countEl.textContent = count;
        if (count === 0) delete cart[name];
        else cart[name].count = count;
        updateCart();
      }
    });
  });

  showCartBtn.addEventListener('click', () => {
    cartModal.style.display = 'flex';
    setTimeout(() => cartModal.classList.add('show'), 10);
  });

  closeCart.addEventListener('click', () => {
    cartModal.classList.remove('show');
    setTimeout(() => cartModal.style.display = 'none', 300);
  });

  // Перед отправкой формы собрать список товаров
  orderForm.addEventListener('submit', (e) => {
    const items = Object.keys(cart).map(name => `${name} x${cart[name].count}`).join(', ');
    orderHidden.value = items + ` | Доставка: 200 ₽ | Итого: ${totalEl.textContent} ₽`;
  });
</script>
